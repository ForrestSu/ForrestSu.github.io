int cost(int k, int T, int egg)
{
    return std::max(F(T-k, egg), F(k-1, egg-1)) + 1;
}

int F(int T, int egg) {
    // only 1 egg, return layers
    if (egg == 1) {
        return T;
    }
    // 0 layer or 1 layer
    if (T == 0 || T == 1) {
        return T;
    }
    //å‘½ä¸­ç¼“å­˜
    if (G_Cache[T][egg]) {
        return G_Cache[T][egg];
    }
    int ans = INF;
    for(int k = 1; k<=T; ++k) {
        int times = cost(k, T, egg);
        ans = std::min(ans, times);
    }
    //åŠ å…¥ç¼“å­˜
    G_Cache[T][egg] = ans;
    return ans;
}
