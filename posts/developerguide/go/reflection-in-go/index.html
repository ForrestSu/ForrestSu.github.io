import (
    "fmt"
    "reflect"
    "testing"
)

type order struct {
    orderId    int
    customName string
}

func reflectStruct(p interface{}) {
    // è·å–ç±»å‹
    tp := reflect.TypeOf(p)
    v := reflect.ValueOf(p)
    fmt.Println("Type ", tp)
    fmt.Println(v)

    k := tp.Kind()
    // Kind = struct Size = 24
    fmt.Println("Kind =", k, "Size =", tp.Size())

    // å¦‚æœä¸ºç»“æ„ä½“ç±»å‹
    if tp.Kind() == reflect.Struct {
        val := reflect.ValueOf(p)
        // è·å–å­—æ®µä¸ªæ•°
        fmt.Println("field count:", val.NumField())
        // éå†æ¯ä¸€ä¸ªå­—æ®µ
        for i := 0; i < val.NumField(); i++ {
            element := val.Field(i)
            fmt.Println("type:", element.Kind(), "value:", element)
            switch element.Kind() {
            case reflect.Int:
                //æå–å¯¹åº”çš„å€¼ï¼Œä¸ºå¯¹åº”çš„ç±»å‹

                fmt.Printf("===> is_string %T %s", str, str)
            default:
                fmt.Println("type is not support:", element.Kind())
            }
        }
    }
}
