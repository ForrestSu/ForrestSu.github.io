1 Preface

Aho-Corasick automation，该算法在1975年产生于贝尔实验室，是著名的多模匹配算法。

AC自动机算法分为3步：(1)构造一棵Trie树，(2)构造Fail指针, (3) 模式匹配过程。

大学时候写的比较多，最近面试被问到了，回答的不完整，这里实现一遍。

#include <iostream>
#include <string>
#include <vector>
#include <cstring>
#include <cassert>
#include <deque>
using namespace std;
#define MAXN 127

struct Trie {
    char ch;
    int is_word; // 0:not-word 1:is-word
    int is_marked; //是否已经计数 0-未计数 1-已计数
    Trie *fail_ptr;
    Trie *next[MAXN];
    int data1;
};

struct Trie* create_node(char ch, int level) {
    Trie *node = (Trie*) malloc(sizeof(Trie));
    memset(node, 0, sizeof(Trie));
    node->ch = ch;
    node->data1 = level;
    node->is_word = 0;
    node->is_marked = 0;
    node->fail_ptr = nullptr;
    return node;
}

bool insert_tree(Trie *root, std::string &words) {
    Trie *cur = root;
    for (size_t i = 0; i < words.length(); ++i) {
        int8_t ch = words[i];
        Trie *child = cur->next[ch];
        if (child == nullptr) {
            child = create_node(ch, i + 1);
            cur->next[ch] = child;
        }
        cur = child;
    }
    cur->is_word = 1;
    return true;
}

/***
 * (1) 按层(BFS搜索)维护失败指针，(失败指针是存在于Tire树上的一个元素节点而已)
 * (2) 每层节点负责维护下一层的失败指针
 * (3) 根节点的失败指针指向nil
 * (4) 根的第一层儿子的失败指针都指向root
 * (5) 每一层怎么为下一层找到失败指针呢?
 *     a. 先拿到一个儿子S，拿出自身的失败指针P；
 *     b. 判断该失败指针P是否也有同名的儿子S'呢？
 *     c. 如果有，则把S的失败指针指向 S',停止；
 *     d. 如果没有，P=P->fail_ptr 上，重复步骤(b)
 */
bool build_fail_ptr(Trie *root) {
    //这个nullptr在后面也作为一个搜索终止条件
    root->fail_ptr = nullptr;
    std::deque<Trie*> Q;
    Q.push_back(root);
    while (!Q.empty()) {
        Trie *cur = Q.front();
        Q.pop_front();
        for (int i = 0; i < MAXN; i++) {
            if (cur->next[i] != nullptr) {
                Trie *child = cur->next[i];
                Q.push_back(child);
                // 为child节点维护失败指针
                if (cur == root) {
                    child->fail_ptr = root;
                    printf("child %c pointer root!\n", child->ch);
                } else {
                    //if layer 2, 3...
                    Trie *ptrfail = cur->fail_ptr;
                    assert(ptrfail != nullptr);
                    // 链式查找
                    while (ptrfail != nullptr) {
                        // same name child occur, got it!
                        if (ptrfail->next[child->ch] != nullptr) {
                            ptrfail = ptrfail->next[child->ch];
                            printf("为[%c]找到失败指针<%c>\n", child->ch, ptrfail->ch);
                            break;
                        }
                        ptrfail = ptrfail->fail_ptr;
                    }
                    //如果没有找到,则把当前元素的失败指针指向
                    child->fail_ptr = (ptrfail == nullptr ? root : ptrfail);
                }
            }
        }
    }
    return true;
}

/**
 * 返回有几种模式串出现过
 */
int search(Trie *root, string matchs) {
    int iCount = 0;
    size_t idx = 0;
    Trie* cur = root;
    while (idx < matchs.length()) {
        char ch = matchs[idx++];
        // match it
        if (cur->next[ch] != nullptr) {
            cur = cur->next[ch];
            if (cur->is_word && cur->is_marked == 0) {
                iCount++;
                cur->is_marked = 1; //marked
                //printf("matchs <%c>\n", ch);
            }
        } else {
           //失配,当前已匹配串,检查其所有后缀串是否出现可能匹配的单词
            while (cur != root) {
                cur = cur->fail_ptr;
                // 针对后缀串计数
                if (cur->is_word && cur->is_marked == 0) {
                    iCount++;
                    cur->is_marked = 1; //marked
                    //printf("suffix matchs <%c>\n", ch);
                }
            }
        }
    }
    printf("search(\"%s\") occur %d.\n", matchs.c_str(), iCount);
    return iCount;
}

void show(Trie *root) {
    // TODO 按层打印
    std::deque<Trie*> Q;
    Q.push_back(root);
    int max_level = -1;
    while (!Q.empty()) {
        Trie *cur = Q.front();
        if (max_level < cur->data1) {
            max_level = cur->data1;
            printf("\nlevel %d\n", max_level);
        }
        printf("%c <word:%d>", cur->ch, cur->is_word);
        Q.pop_front();
        for (int i = 0; i < MAXN; ++i) {
            if (cur->next[i] != nullptr) {
                Q.push_back(cur->next[i]);
            }
        }
    }
    printf("\n");
}

int main() {
    printf("%lld\n", sizeof(Trie));
    Trie *root = create_node('r', 0);
    {
        string words = "she";
        insert_tree(root, words);
    }
    {
        string words = "he";
        insert_tree(root, words);
    }
    show(root);
    build_fail_ptr(root);
    {
        string words = "sher";
        search(root, words);
    }
    return 0;
}
